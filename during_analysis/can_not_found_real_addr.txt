function setCookie(name, value)
{
	var Days = 90;
	var exp  = new Date();
	exp.setTime(exp.getTime() + Days*24*60*60*1000);
	document.cookie = name + "="+ escape (value) + ";expires=" + exp.toGMTString();

}

function getNowStamp()
{
	var timestamp = Date.parse(new Date());
	return parseInt(timestamp/1000);
}

function commonReg(cont)
{
		if(cont.length < 1){
			return 'blank';
		}
		var pattern = /\\.com|\\.cn|\\.net|\\.org|\\.biz|\\.info|\\.cn|\\.mobi|\\.name|\\.sh|\\.ac|\\.la|\\.travel|\\.tm|\\.us|\\.cc |\\.tv| \\.mn| \\.saia| \\.hn|\\.lc |\\.hk|\\.bz|\\.ws |\\.tel|\\.cm|\\.tw|\\.io|\\.ac/gi;
		var stpattern = /songtaste.com/i;
		if (!stpattern.test(cont)) {
			if (pattern.test(cont)) {
				return 'link';
			}
		}

		var par = /\\[.*?\\].*?\\[\\/.*?\\]*?]/i;
		if (par.test(cont)) {
			return 'stab';
		}

		var reg = new RegExp('[^<>]*?>|Uis');
	  	if (reg.test(cont)) {
			return 'tab';
	  	}
		return '';
}

function saveSongName()
{
	var songname = $('#songname').val();
	var singername = $('#singername').val();
	if(songname == '' || songname == '你认为正确的') {
		alert('歌曲名称不能为空');
		return false;
	}
	if(singername == '') {
		alert('歌手名不能为空');
		return false;
	}
	var regrs = commonReg(songname);
	switch(regrs) {
		case 'link':
			alert('歌名中不能存在链接');
			return false;break;
		case 'stab':
			alert('歌名中不允许有自定义标签');
			return false;break;
		case 'tab':
			alert('歌名中不允许有标签');
			return false;break;
		default:
			break;
	}
	var regrs = commonReg(singername);
	switch(regrs) {
		case 'link':
			alert('歌手名中不能存在链接');
			return false;break;
		case 'stab':
			alert('歌手名中不允许有自定义标签');
			return false;break;
		case 'tab':
			alert('歌手名中不允许有标签');
			return false;break;
		default:
			break;
	}
	
	var songid = 2428041;
	var operurl = "/info_oper.php?tag=addsongname&songid="+songid+"&songname="+songname+"&singername="+singername;
	$.ajax({
		type : "GET",
		url : operurl,
		contentType : "text/html;charset=gb2312",
		success : function(msg){
			if(msg == 'nologin'){
				alert('对不起，您需要登录才能添加歌曲名');
			}
			else if(msg == 'userStatError') {
				alert('您的帐户已经被冻结，不能做任何操作');
			}
			else if(msg == 'nosongname') {
				alert('歌曲名不能为空');
			}
			else if(msg == 'exceedlimit') {
				alert('添加的歌曲名不能超过3个');
			}
			else if(msg == 'userlimit') {
				alert('您只能为该歌曲添加一个歌曲名');
			}
			else if(msg == 'existed') {
				alert('您添加的歌名已存在');
			}
			else if(msg == 'getNameError') {
				alert('获取歌名列表失败，请刷新');
			}
			else{
				$('ul.has_hb').html(msg);
			}
		}
	});
}

function favSongName(nameid)
{
	var favsonglist = getCookie('favSongName');
	if(favsonglist == null){
		favsonglist = '';
	}
	if(favsonglist.length > 0 && favsonglist.indexOf(nameid)>-1) {
		alert('您已经顶过该歌曲名');
		return false;
	}
	var operurl = "/info_oper.php?tag=favsongname&nameid="+nameid;
	$.ajax({
		type : "GET",
		url : operurl,
		success : function(msg){
			if(msg == 'nologin'){
				alert('对不起，您需要登录才能添加歌曲名');
			}
			else if(msg == 'userStatError') {
				alert('您的帐户已经被冻结，不能做任何操作');
			}
			else{
				$('#favname_'+nameid).html('顶'+msg);
				if(favsonglist != '') {
					favsonglist += ',';
				}
				favsonglist += nameid;
				setCookie('favSongName', favsonglist);
			}
		}
	});
}

function showLightbox() {
	var login_status = 0;
	if(!checkLogin()) {
		login_status = 1;
	}
	else {
		var ses = getCookie("CookDmID");
		if(ses && ses >0) {
			return;
		}
		else if(ses < 0){
			login_status = 2;
		}
		else {
			$.ajax({
				url: "/info_oper.php?tag=checkbind",
				async: false,
				success : function(data) {
					if(data==1){
					   login_status = 0;
					   return;
					}
					login_status = 2;
				}
			});
		}
	}
	if(login_status == 0) return false;
	if(login_status == 1) {
		var t = setTimeout(showLoginLightbox,30000);
		return t;
	}
	if(login_status == 2) {
		var t = setTimeout(showBindLightbox,30000);
		return t;
	}
}

		var songid = $("#song_id").val();
		if(!songid) {
			return false;
		}
//		var secoder = $('#secoder').val();

		var cmtcont = $("#cont").val();
		if(cmtcont.length < 1){
			alert("评论内容不能为空");
			return false;
		}
		var pattern = /\\.com|\\.cn|\\.net|\\.org|\\.biz|\\.info|\\.cn|\\.mobi|\\.name|\\.sh|\\.ac|\\.la|\\.travel|\\.tm|\\.us|\\.cc |\\.tv| \\.mn| \\.saia| \\.hn|\\.lc |\\.hk|\\.bz|\\.ws |\\.tel|\\.cm|\\.tw|\\.io|\\.ac/gi;
		var stpattern = /songtaste.com/i;
		if (!stpattern.test(cmtcont)) {
			if (pattern.test(cmtcont)) {
				alert("评论内容不允许有链接！");
				document.getElementById("cont").focus();
				return(false);
			}
		}

		var par = /\\[.*?\\].*?\\[\\/.*?\\]*?]/i;
		if (par.test(cmtcont)) {
			alert("不允许有自定义标签出现！");
			document.getElementById("cont").focus();
			return(false);
		}

		var reg = new RegExp('[^<>]*?>|Uis');
	  	if (reg.test(cmtcont)) {
			alert("评论内容不允许有标签！");
			document.getElementById("cont").focus();
			return(false);
	   
	  	}
	  	
	  	var uid = getCookie("CookID");
	  	if(!uid) {
	  		alert('对不起，您需要登录才能评论');
	  		return false;
	  	}
	  	else 
	  	{
		  	var lastcmt = getCookie('LastCmt'+uid);
		  	var nowtime = getNowStamp();
		  	if(!lastcmt) 
		  	{
		  		setCookie("LastCmt"+uid, nowtime);
		  	} 
		  	else 
		  	{
		  		if((nowtime - lastcmt) < 30) 
		  		{
		  			alert('对不起，您留言太快了，两次留言时间不能低于30秒！');
		  			return false;
		  		}
		  	}
	  	}
	  	
		var cmturl = "/info_oper.php?tag=song_cmt&songid="+songid+"&cont="+cmtcont;
		$.ajax({
			type : "GET",
			url : cmturl,
			contentType : "text/html;charset=gbk",
			beforeSend : function(){
				document.getElementById("cont").disabled = true;
				document.getElementById("subcmt").disabled = true;
				var promptHtml = "<div id='cmt_prompt' style='text-align:center;margin:0 0 20px 0;clear:both;'>正在发送请求，请稍候…</div>";
				$('div.cmt_list').before(promptHtml);
			},
			
			success : function(msg){
				document.getElementById("cont").value = '';
				document.getElementById("cont").disabled = false;
				document.getElementById("subcmt").disabled = false;
				if(document.getElementById("cmt_prompt")) {
					$('#cmt_prompt').remove();
				}
				
				if(msg == 'nologin'){
					alert('对不起，您需要登录才能评论');
				}
				else if(msg == 'userStatError') {
					alert('您的帐户已经被冻结，不能做任何操作');
				}
				else if(msg == 'getcmtError'){
					alert('获取评论列表失败，请刷新');
				}
				else if(msg == 'checkCmtDt') {
					alert('对不起，您留言太快了，两次留言时间不能低于30秒！');
				}
				else if(msg == 'regtimelimit') {
					alert('对不起，该用户(上传者)设置了注册时间超过一定天数的会员才能留言！');
				}
				else if(msg == 'badfred') {
					alert('对方用户已经把您列为坏蛋名单中，您不能给对方以及对方歌曲专辑留言！');
				}
				else if(msg == 'regtime10') {
					alert('注册10天后才能对歌曲留言！');
				}
				else if(msg == 'secoderror') {
					alert('您的验证码错误！');
				}
				else {
					$("div.cmt_list").html(msg);
//					$("#secoder").val('');
//					$("#secoder_img a").trigger("click");
				}
			},
			error : function(){
				document.getElementById("cont").disabled = false;
				document.getElementById("subcmt").disabled = false;
				if(document.getElementById("cmt_prompt")) {
					$('#cmt_prompt').remove();
				}
				alert('发布留言失败，请重试！');
			}
		}); 
	}

function ratesong(){

		$.ajax({
			type : "GET",

			success : function(msg){
				switch(msg){
					case 'nologin':
						alert('对不起，您需要登录才能对此歌曲进行评分');break;
					case 'rated':
						alert('您以前已经认为这首歌好听了,不能重复评价');break;
					case 'timelimit':
						alert('对不起，两次歌曲评分至少间隔1分钟');break;
					case 'norateself':
						alert('对不起，您不能给自己首次推荐的歌曲评分');break;
					case 'rateok':
						alert('您的评分已经生效，感谢您对歌曲的支持');break;
				}
			}
		});
	}
